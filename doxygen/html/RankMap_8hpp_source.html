42 size_t find(typename Slice<F>::Location const& p) const noexcept {
43 if (RANK_ROUND_ROBIN) {
44 return p.source * np + p.rank;
45 } else {
46 const size_t
47 rankPosition = p.source * clusterInfo.ranksPerNode
48 + clusterInfo.rankInfos[p.rank].localRank
49 ;
50 return rankPosition * clusterInfo.nNodes
51 + clusterInfo.rankInfos[p.rank].nodeId
52 ;
53 }
54 }
55
56 size_t nSources() const noexcept {
57 return size / np + size_t(size % np != 0);
58 }
59
60
61 bool isPaddingRank(size_t rank) const noexcept {
62 return size % np == 0
63 ? false
64 : rank > (size % np - 1)
65 ;
66 }
67
68 bool isSourcePadding(size_t rank, size_t source) const noexcept {
69 return source == nSources() && isPaddingRank(rank);
70 }

72 typename Slice<F>::Location
73 find(ABCTuple const& abc, typename Slice<F>::Type sliceType) const {
74 // tuple = {11, 8} when abc = {11, 8, 9} and sliceType = AB
75 // tuple = {11, 0} when abc = {11, 8, 9} and sliceType = A
76 const auto tuple = Slice<F>::subtupleBySlice(abc, sliceType);
77
78 const size_t index
79 = tuple[0]
80 + tuple[1] * (lengths.size() > 1 ? lengths[0] : 0)
81 ;
82
83 size_t rank, source;
84
85 if (RANK_ROUND_ROBIN) {
86
87 rank = index % np;
88 source = index / np;
89
90 } else {
91
92 size_t const
93
94 // the node that will be assigned to
95 nodeId = index % clusterInfo.nNodes
96
97 // how many times it has been assigned to the node
98 , s_n = index / clusterInfo.nNodes
99
100 // which local rank in the node should be
101 , localRank = s_n % clusterInfo.ranksPerNode
102
103 // and the local source (how many times we chose this local rank)
104 , localSource = s_n / clusterInfo.ranksPerNode
105 ;
106
107 // find the localRank-th entry in clusterInfo
108 auto const& it =
109 std::find_if(clusterInfo.rankInfos.begin(),
110 clusterInfo.rankInfos.end(),
111 [nodeId, localRank](RankInfo const& ri) {
112 return ri.nodeId == nodeId
113 && ri.localRank == localRank
114 ;
115 });
116 if (it == clusterInfo.rankInfos.end()) {
117 throw "FATAL! Error in node distribution of the slices";
118 }
119
120 rank = (*it).globalRank;
121 source = localSource;
122
123 }
124
125 return
126 { rank
127 , source
128 };
129 }
130
131 };

133}
