23 template <typename F=double>
24 double getEnergyDistinct
25 ( const F epsabc
26 , std::vector<F> const& epsi
27 , std::vector<F> const& Tijk_
28 , std::vector<F> const& Zijk_
29 ) {
30 constexpr size_t blockSize=16;
31 F energy(0.);
32 const size_t No = epsi.size();
33 for (size_t kk=0; kk<No; kk+=blockSize){
34 const size_t kend( std::min(No, kk+blockSize) );
35 for (size_t jj(kk); jj<No; jj+=blockSize){
36 const size_t jend( std::min( No, jj+blockSize) );
37 for (size_t ii(jj); ii<No; ii+=blockSize){
38 const size_t iend( std::min( No, ii+blockSize) );
39 for (size_t k(kk); k < kend; k++){
40 const F ek(epsi[k]);
41 const size_t jstart = jj > k ? jj : k;
42 for (size_t j(jstart); j < jend; j++){
43 F const ej(epsi[j]);
44 F const facjk = j == k ? F(0.5) : F(1.0);
45 size_t istart = ii > j ? ii : j;
46 for (size_t i(istart); i < iend; i++){
47 const F
48 ei(epsi[i])
49 , facij = i == j ? F(0.5) : F(1.0)
50 , denominator(epsabc - ei - ej - ek)
51 , U(Zijk_[i + No*j + No*No*k])
52 , V(Zijk_[i + No*k + No*No*j])
53 , W(Zijk_[j + No*i + No*No*k])
54 , X(Zijk_[j + No*k + No*No*i])
55 , Y(Zijk_[k + No*i + No*No*j])
56 , Z(Zijk_[k + No*j + No*No*i])
57 , A(maybeConjugate<F>(Tijk_[i + No*j + No*No*k]))
58 , B(maybeConjugate<F>(Tijk_[i + No*k + No*No*j]))
59 , C(maybeConjugate<F>(Tijk_[j + No*i + No*No*k]))
60 , D(maybeConjugate<F>(Tijk_[j + No*k + No*No*i]))
61 , E(maybeConjugate<F>(Tijk_[k + No*i + No*No*j]))
62 , _F(maybeConjugate<F>(Tijk_[k + No*j + No*No*i]))
63 , value
64 = 3.0 * ( A * U
65 + B * V
66 + C * W
67 + D * X
68 + E * Y
69 + _F * Z )
70 + ( ( U + X + Y )
71 - 2.0 * ( V + W + Z )
72 ) * ( A + D + E )
73 + ( ( V + W + Z )
74 - 2.0 * ( U + X + Y )
75 ) * ( B + C + _F )
76 ;
77 energy += 2.0 * value / denominator * facjk * facij;
78 } // i
79 } // j
80 } // k
81 } // ii
82 } // jj
83 } // kk
84 return std::real(energy);
85 }

101 const size_t jend( std::min( jj+blockSize, No) );

104 for (size_t k(kk); k < kend; k++){
105 const F ek(epsi[k]);
106 const size_t jstart = jj > k ? jj : k;
107 for(size_t j(jstart); j < jend; j++){
108 const F facjk( j == k ? F(0.5) : F(1.0));
109 const F ej(epsi[j]);
110 const size_t istart = ii > j ? ii : j;
111 for(size_t i(istart); i < iend; i++){
112 const F
113 ei(epsi[i])
114 , facij ( i==j ? F(0.5) : F(1.0))
115 , denominator(epsabc - ei - ej - ek)
116 , U(Zijk_[i + No*j + No*No*k])
117 , V(Zijk_[j + No*k + No*No*i])
118 , W(Zijk_[k + No*i + No*No*j])
119 , A(maybeConjugate<F>(Tijk_[i + No*j + No*No*k]))
120 , B(maybeConjugate<F>(Tijk_[j + No*k + No*No*i]))
121 , C(maybeConjugate<F>(Tijk_[k + No*i + No*No*j]))
122 , value
123 = F(3.0) * ( A * U
124 + B * V
125 + C * W
126 )
127 - ( A + B + C ) * ( U + V + W )
128 ;
129 energy += F(2.0) * value / denominator * facjk * facij;
130 } // i
131 } // j
132 } // k
133 } // ii
134 } // jj
135 } // kk
