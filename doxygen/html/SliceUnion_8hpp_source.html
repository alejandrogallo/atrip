30  /*
31  * This function should enforce an important property of a SliceUnion.
32  * Namely, there can be no two Slices of the same nature.
33  *
34  * This means that there can be at most one slice with a given Ty_x_Tu.
35  */
36  void checkForDuplicates() const {
37  std::vector<typename Slice<F>::Ty_x_Tu> tytus;
38  for (auto const& s: slices) {
39  if (s.isFree()) continue;
40  tytus.push_back({s.info.type, s.info.tuple});
41  }
42 
43  for (auto const& tytu: tytus) {
44  if (std::count(tytus.begin(), tytus.end(), tytu) > 1)
45  throw "Invariance violated, more than one slice with same Ty_x_Tu";
46  }

57  return std::make_pair(type, tuple);

62  /* buildLocalDatabase
63  *
64  * It should build a database of slices so that we know what is needed
65  * to fetch in the next iteration represented by the tuple 'abc'.
66  *
67  * 1. The algorithm works as follows, we build a database of the all
68  * the slice types that we need together with their tuple.
69  *
70  * 2. Look in the SliceUnion if we already have this tuple,
71  * if we already have it mark it (TODO)
72  *
73  * 3. If we don't have the tuple, look for a (state=acceptor, type=blank)
74  * slice and mark this slice as type=Fetch with the corresponding type
75  * and tuple.
76  *
77  * NOTE: The algorithm should certify that we always have enough blank
78  * slices.
79  *
80  */
81  typename
82  Slice<F>::LocalDatabase buildLocalDatabase(ABCTuple const& abc) {
83  typename Slice<F>::LocalDatabase result;
84 
85  auto const needed = neededSlices(abc);
86 
87  WITH_RANK << "__db__:needed:" << pretty_print(needed) << "\n";
88  // BUILD THE DATABASE
89  // we need to loop over all sliceTypes that this TensorUnion
90  // is representing and find out how we will get the corresponding
91  // slice for the abc we are considering right now.
92  for (auto const& pair: needed) {
93  auto const type = pair.first;
94  auto const tuple = pair.second;
95  auto const from = rankMap.find(abc, type);
96 
97 #ifdef HAVE_OCD
98  WITH_RANK << "__db__:want:" << pretty_print(pair) << "\n";
99  for (auto const& s: slices)
100  WITH_RANK << "__db__:guts:ocd "
101  << s.info << " pt " << s.data
102  << "\n";
103 #endif
104 
105 #ifdef HAVE_OCD
106  WITH_RANK << "__db__: checking if exact match" << "\n";
107 #endif
108  {
109  // FIRST: look up if there is already a *Ready* slice matching what we
110  // need
111  auto const& it
112  = std::find_if(slices.begin(), slices.end(),
113  [&tuple, &type](Slice<F> const& other) {
114  return other.info.tuple == tuple
115  && other.info.type == type
116  // we only want another slice when it
117  // has already ready-to-use data
118  && other.isUnwrappable()
119  ;
120  });
121  if (it != slices.end()) {
122  // if we find this slice, it means that we don't have to do anything
123  WITH_RANK << "__db__: EXACT: found EXACT in name=" << name
124  << " for tuple " << tuple[0] << ", " << tuple[1]
125  << " ptr " << it->data
126  << "\n";
127  result.push_back({name, it->info});
128  continue;
129  }
130  }
131 
132 #ifdef HAVE_OCD
133  WITH_RANK << "__db__: checking if recycle" << "\n";
134 #endif
135  // Try to find a recyling possibility ie. find a slice with the same
136  // tuple and that has a valid data pointer.
137  auto const& recycleIt
138  = std::find_if(slices.begin(), slices.end(),
139  [&tuple, &type](Slice<F> const& other) {
140  return other.info.tuple == tuple
141  && other.info.type != type
142  && other.isRecyclable()
143  ;
144  });
145 
146  // if we find this recylce, then we find a Blank slice
147  // (which should exist by construction :THINK)
148  //
149  if (recycleIt != slices.end()) {
150  auto& blank = Slice<F>::findOneByType(slices, Slice<F>::Blank);
151  // TODO: formalize this through a method to copy information
152  // from another slice
153  blank.data = recycleIt->data;
154  blank.info.type = type;
155  blank.info.tuple = tuple;
156  blank.info.state = Slice<F>::Recycled;
157  blank.info.from = from;
158  blank.info.recycling = recycleIt->info.type;
159  result.push_back({name, blank.info});
160  WITH_RANK << "__db__: RECYCLING: n" << name
161  << " " << pretty_print(abc)
162  << " get " << pretty_print(blank.info)
163  << " from " << pretty_print(recycleIt->info)
164  << " ptr " << recycleIt->data
165  << "\n"
166  ;
167  continue;
168  }
169 
170  // in this case we have to create a new slice
171  // this means that we should have a blank slice at our disposal
172  // and also the freePointers should have some elements inside,
173  // so we pop a data pointer from the freePointers container
174 #ifdef HAVE_OCD
175  WITH_RANK << "__db__: none work, doing new" << "\n";
176 #endif
177  {
178  WITH_RANK << "__db__: NEW: finding blank in " << name
179  << " for type " << type
180  << " for tuple " << tuple[0] << ", " << tuple[1]
181  << "\n"
182  ;
183  auto& blank = Slice<F>::findOneByType(slices, Slice<F>::Blank);
184  blank.info.type = type;
185  blank.info.tuple = tuple;
186  blank.info.from = from;
187 
188  // Handle self sufficiency
189  blank.info.state = Atrip::rank == from.rank
190  ? Slice<F>::SelfSufficient
191  : Slice<F>::Fetch
192  ;
193  if (blank.info.state == Slice<F>::SelfSufficient) {
194  blank.data = sources[from.source].data();
195  } else {
196  if (freePointers.size() == 0) {
197  std::stringstream stream;
198  stream << "No more free pointers "
199  << "for type " << type
200  << " and name " << name
201  ;
202  throw std::domain_error(stream.str());
203  }
204  auto dataPointer = freePointers.begin();
205  freePointers.erase(dataPointer);
206  blank.data = *dataPointer;
207  }
208 
209  result.push_back({name, blank.info});
210  continue;
211  }

216  for (auto const& s: slices)
217  WITH_RANK << "__db__:guts:ocd:__end__ " << s.info << "\n";
218 #endif

225  /*
226  * Garbage collect slices not needed for the next iteration.
227  *
228  * It will throw if it tries to gc a slice that has not been
229  * previously unwrapped, as a safety mechanism.
230  */
231  void clearUnusedSlicesForNext(ABCTuple const& abc) {
232  auto const needed = neededSlices(abc);
233 
234  // CLEAN UP SLICES, FREE THE ONES THAT ARE NOT NEEDED ANYMORE
235  for (auto& slice: slices) {
236  // if the slice is free, then it was not used anyways
237  if (slice.isFree()) continue;
238 
239 
240  // try to find the slice in the needed slices list
241  auto const found
242  = std::find_if(needed.begin(), needed.end(),
243  [&slice] (typename Slice<F>::Ty_x_Tu const& tytu) {
244  return slice.info.tuple == tytu.second
245  && slice.info.type == tytu.first
246  ;
247  });
248 
249  // if we did not find slice in needed, then erase it
250  if (found == needed.end()) {
251 
252  // We have to be careful about the data pointer,
253  // for SelfSufficient, the data pointer is a source pointer
254  // of the slice, so we should just wipe it.
255  //
256  // For Ready slices, we have to be careful if there are some
257  // recycled slices depending on it.
258  bool freeSlicePointer = true;
259 
260  // allow to gc unwrapped and recycled, never Fetch,
261  // if we have a Fetch slice then something has gone very wrong.
262  if (!slice.isUnwrapped() && slice.info.state != Slice<F>::Recycled)
263  throw
264  std::domain_error("Trying to garbage collect "
265  " a non-unwrapped slice! "
266  + pretty_print(&slice)
267  + pretty_print(slice.info));
268 
269  // it can be that our slice is ready, but it has some hanging
270  // references lying around in the form of a recycled slice.
271  // Of course if we need the recycled slice the next iteration
272  // this would be fatal, because we would then free the pointer
273  // of the slice and at some point in the future we would
274  // overwrite it. Therefore, we must check if slice has some
275  // references in slices and if so then
276  //
277  // - we should mark those references as the original (since the data
278  // pointer should be the same)
279  //
280  // - we should make sure that the data pointer of slice
281  // does not get freed.
282  //
283  if (slice.info.state == Slice<F>::Ready) {
284  WITH_OCD WITH_RANK
285  << "__gc__:" << "checking for data recycled dependencies\n";
286  auto recycled
287  = Slice<F>::hasRecycledReferencingToIt(slices, slice.info);
288  if (recycled.size()) {
289  Slice<F>* newReady = recycled[0];
290  WITH_OCD WITH_RANK
291  << "__gc__:" << "swaping recycled "
292  << pretty_print(newReady->info)
293  << " and "
294  << pretty_print(slice.info)
295  << "\n";
296  newReady->markReady();
297  assert(newReady->data == slice.data);
298  freeSlicePointer = false;
299 
300  for (size_t i = 1; i < recycled.size(); i++) {
301  auto newRecyled = recycled[i];
302  newRecyled->info.recycling = newReady->info.type;
303  WITH_OCD WITH_RANK
304  << "__gc__:" << "updating recycled "
305  << pretty_print(newRecyled->info)
306  << "\n";
307  }
308 
309  }
310  }
311 
312  // if the slice is self sufficient, do not dare touching the
313  // pointer, since it is a pointer to our sources in our rank.
314  if ( slice.info.state == Slice<F>::SelfSufficient
315  || slice.info.state == Slice<F>::Recycled
316  ) {
317  freeSlicePointer = false;
318  }
319 
320  // make sure we get its data pointer to be used later
321  // only for non-recycled, since it can be that we need
322  // for next iteration the data of the slice that the recycled points
323  // to
324  if (freeSlicePointer) {
325  freePointers.insert(slice.data);
326  WITH_RANK << "~~~:cl(" << name << ")"
327  << " added to freePointer "
328  << pretty_print(freePointers)
329  << "\n";
330  } else {
331  WITH_OCD WITH_RANK << "__gc__:not touching the free Pointer\n";
332  }
333 
334  // at this point, let us blank the slice
335  WITH_RANK << "~~~:cl(" << name << ")"
336  << " freeing up slice "
337  // TODO: make this possible because of Templates
338  // TODO: there is a deduction error here
339  // << " info " << slice.info
340  << "\n";
341  slice.free();
342  }
343 
344  }

498  F* unwrapSlice(typename Slice<F>::Type type, ABCTuple const& abc) {
499  WITH_CRAZY_DEBUG
500  WITH_RANK << "__unwrap__:slice " << type << " w n "
501  << name
502  << " abc" << pretty_print(abc)
503  << "\n";
504  auto& slice = Slice<F>::findByTypeAbc(slices, type, abc);
505  //WITH_RANK << "__unwrap__:info " << slice.info << "\n";
506  switch (slice.info.state) {
507  case Slice<F>::Dispatched:
508  WITH_RANK << "__unwrap__:Fetch: " << &slice
509  << " info " << pretty_print(slice.info)
510  << "\n";
511  slice.unwrapAndMarkReady();
512  return slice.data;
513  break;
514  case Slice<F>::SelfSufficient:
515  WITH_RANK << "__unwrap__:SelfSufficient: " << &slice
516  << " info " << pretty_print(slice.info)
517  << "\n";
518  return slice.data;
519  break;
520  case Slice<F>::Ready:
521  WITH_RANK << "__unwrap__:READY: UNWRAPPED ALREADY" << &slice
522  << " info " << pretty_print(slice.info)
523  << "\n";
524  return slice.data;
525  break;
526  case Slice<F>::Recycled:
527  WITH_RANK << "__unwrap__:RECYCLED " << &slice
528  << " info " << pretty_print(slice.info)
529  << "\n";
530  return unwrapSlice(slice.info.recycling, abc);
531  break;
532  case Slice<F>::Fetch:
533  case Slice<F>::Acceptor:
534  throw std::domain_error("Can't unwrap an acceptor or fetch slice!");
535  break;
536  default:
537  throw std::domain_error("Unknown error unwrapping slice!");
538  }
539  return slice.data;
540  }
