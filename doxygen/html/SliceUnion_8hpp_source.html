33 /*
34 * This function should enforce an important property of a SliceUnion.
35 * Namely, there can be no two Slices of the same nature.
36 *
37 * This means that there can be at most one slice with a given Ty_x_Tu.
38 */
39 void checkForDuplicates() const {
40 std::vector<typename Slice<F>::Ty_x_Tu> tytus;
41 for (auto const& s: slices) {
42 if (s.isFree()) continue;
43 tytus.push_back({s.info.type, s.info.tuple});
44 }
45
46 for (auto const& tytu: tytus) {
47 if (std::count(tytus.begin(), tytus.end(), tytu) > 1)
48 throw "Invariance violated, more than one slice with same Ty_x_Tu";
49 }
50
51 }

53 std::vector<typename Slice<F>::Ty_x_Tu> neededSlices(ABCTuple const& abc) {
54 std::vector<typename Slice<F>::Ty_x_Tu> needed(sliceTypes.size());
55 // build the needed vector
56 std::transform(sliceTypes.begin(), sliceTypes.end(),
57 needed.begin(),
58 [&abc](typename Slice<F>::Type const type) {
59 auto tuple = Slice<F>::subtupleBySlice(abc, type);
60 return std::make_pair(type, tuple);
61 });
62 return needed;
63 }

65 /* buildLocalDatabase
66 *
67 * It should build a database of slices so that we know what is needed
68 * to fetch in the next iteration represented by the tuple 'abc'.
69 *
70 * 1. The algorithm works as follows, we build a database of the all
71 * the slice types that we need together with their tuple.
72 *
73 * 2. Look in the SliceUnion if we already have this tuple,
74 * if we already have it mark it (TODO)
75 *
76 * 3. If we don't have the tuple, look for a (state=acceptor, type=blank)
77 * slice and mark this slice as type=Fetch with the corresponding type
78 * and tuple.
79 *
80 * NOTE: The algorithm should certify that we always have enough blank
81 * slices.
82 *
83 */
84 typename
85 Slice<F>::LocalDatabase buildLocalDatabase(ABCTuple const& abc) {
86 typename Slice<F>::LocalDatabase result;
87
88 auto const needed = neededSlices(abc);
89
90 WITH_RANK << "__db__:needed:" << pretty_print(needed) << "\n";
91 // BUILD THE DATABASE
92 // we need to loop over all sliceTypes that this TensorUnion
93 // is representing and find out how we will get the corresponding
94 // slice for the abc we are considering right now.
95 for (auto const& pair: needed) {
96 auto const type = pair.first;
97 auto const tuple = pair.second;
98 auto const from = rankMap.find(abc, type);
99
100#ifdef HAVE_OCD
101 WITH_RANK << "__db__:want:" << pretty_print(pair) << "\n";
102 for (auto const& s: slices)
103 WITH_RANK << "__db__:guts:ocd "
104 << s.info << " pt " << s.data
105 << "\n";
106#endif
107
108#ifdef HAVE_OCD
109 WITH_RANK << "__db__: checking if exact match" << "\n";
110#endif
111 {
112 // FIRST: look up if there is already a *Ready* slice matching what we
113 // need
114 auto const& it
115 = std::find_if(slices.begin(), slices.end(),
116 [&tuple, &type](Slice<F> const& other) {
117 return other.info.tuple == tuple
118 && other.info.type == type
119 // we only want another slice when it
120 // has already ready-to-use data
121 && other.isUnwrappable()
122 ;
123 });
124 if (it != slices.end()) {
125 // if we find this slice, it means that we don't have to do anything
126 WITH_RANK << "__db__: EXACT: found EXACT in name=" << name
127 << " for tuple " << tuple[0] << ", " << tuple[1]
128 << " ptr " << it->data
129 << "\n";
130 result.push_back({name, it->info});
131 continue;
132 }
133 }
134
135#ifdef HAVE_OCD
136 WITH_RANK << "__db__: checking if recycle" << "\n";
137#endif
138 // Try to find a recyling possibility ie. find a slice with the same
139 // tuple and that has a valid data pointer.
140 auto const& recycleIt
141 = std::find_if(slices.begin(), slices.end(),
142 [&tuple, &type](Slice<F> const& other) {
143 return other.info.tuple == tuple
144 && other.info.type != type
145 && other.isRecyclable()
146 ;
147 });
148
149 // if we find this recylce, then we find a Blank slice
150 // (which should exist by construction :THINK)
151 //
152 if (recycleIt != slices.end()) {
153 auto& blank = Slice<F>::findOneByType(slices, Slice<F>::Blank);
154 // TODO: formalize this through a method to copy information
155 // from another slice
156 blank.data = recycleIt->data;
157 blank.info.type = type;
158 blank.info.tuple = tuple;
159 blank.info.state = Slice<F>::Recycled;
160 blank.info.from = from;
161 blank.info.recycling = recycleIt->info.type;
162 result.push_back({name, blank.info});
163 WITH_RANK << "__db__: RECYCLING: n" << name
164 << " " << pretty_print(abc)
165 << " get " << pretty_print(blank.info)
166 << " from " << pretty_print(recycleIt->info)
167 << " ptr " << recycleIt->data
168 << "\n"
169 ;
170 continue;
171 }
172
173 // in this case we have to create a new slice
174 // this means that we should have a blank slice at our disposal
175 // and also the freePointers should have some elements inside,
176 // so we pop a data pointer from the freePointers container
177#ifdef HAVE_OCD
178 WITH_RANK << "__db__: none work, doing new" << "\n";
179#endif
180 {
181 WITH_RANK << "__db__: NEW: finding blank in " << name
182 << " for type " << type
183 << " for tuple " << tuple[0] << ", " << tuple[1]
184 << "\n"
185 ;
186 auto& blank = Slice<F>::findOneByType(slices, Slice<F>::Blank);
187 blank.info.type = type;
188 blank.info.tuple = tuple;
189 blank.info.from = from;
190
191 // Handle self sufficiency
192 blank.info.state = Atrip::rank == from.rank
193 ? Slice<F>::SelfSufficient
194 : Slice<F>::Fetch
195 ;
196 if (blank.info.state == Slice<F>::SelfSufficient) {
197#if defined(HAVE_CUDA)
198 const size_t _size = sizeof(F) * sources[from.source].size();
199 // TODO: this is code duplication with downstairs
200 if (freePointers.size() == 0) {
201 std::stringstream stream;
202 stream << "No more free pointers "
203 << "for type " << type
204 << " and name " << name
205 ;
206 throw std::domain_error(stream.str());
207 }
208 auto dataPointer = freePointers.begin();
209 freePointers.erase(dataPointer);
210 blank.data = *dataPointer;
211 //
212 //
213 // TODO [#A]: do cuMemcpy of
214 // sources[from.source].data() ⇒ blank.data
215 // Do this when everything else is working.
216 // This will probably be a bottleneck of the H-to-D communication,
217 // as most slices are SelfSufficient.
218 //
219 //
220#else
221 blank.data = sources[from.source].data();
222#endif
223 } else {
224 if (freePointers.size() == 0) {
225 std::stringstream stream;
226 stream << "No more free pointers "
227 << "for type " << type
228 << " and name " << name
229 ;
230 throw std::domain_error(stream.str());
231 }
232 auto dataPointer = freePointers.begin();
233 freePointers.erase(dataPointer);
234 blank.data = *dataPointer;
235 }
236
237 result.push_back({name, blank.info});
238 continue;
239 }
240
241 }
242
243#ifdef HAVE_OCD
244 for (auto const& s: slices)
245 WITH_RANK << "__db__:guts:ocd:__end__ " << s.info << "\n";
246#endif
247
248
249 return result;
250
251 }

253 /*
254 * Garbage collect slices not needed for the next iteration.
255 *
256 * It will throw if it tries to gc a slice that has not been
257 * previously unwrapped, as a safety mechanism.
258 */
259 void clearUnusedSlicesForNext(ABCTuple const& abc) {
260 auto const needed = neededSlices(abc);
261
262 // CLEAN UP SLICES, FREE THE ONES THAT ARE NOT NEEDED ANYMORE
263 for (auto& slice: slices) {
264 // if the slice is free, then it was not used anyways
265 if (slice.isFree()) continue;
266
267
268 // try to find the slice in the needed slices list
269 auto const found
270 = std::find_if(needed.begin(), needed.end(),
271 [&slice] (typename Slice<F>::Ty_x_Tu const& tytu) {
272 return slice.info.tuple == tytu.second
273 && slice.info.type == tytu.first
274 ;
275 });
276
277 // if we did not find slice in needed, then erase it
278 if (found == needed.end()) {
279
280 // We have to be careful about the data pointer,
281 // for SelfSufficient, the data pointer is a source pointer
282 // of the slice, so we should just wipe it.
283 //
284 // For Ready slices, we have to be careful if there are some
285 // recycled slices depending on it.
286 bool freeSlicePointer = true;
287
288 // allow to gc unwrapped and recycled, never Fetch,
289 // if we have a Fetch slice then something has gone very wrong.
290 if (!slice.isUnwrapped() && slice.info.state != Slice<F>::Recycled)
291 throw
292 std::domain_error("Trying to garbage collect "
293 " a non-unwrapped slice! "
294 + pretty_print(&slice)
295 + pretty_print(slice.info));
296
297 // it can be that our slice is ready, but it has some hanging
298 // references lying around in the form of a recycled slice.
299 // Of course if we need the recycled slice the next iteration
300 // this would be fatal, because we would then free the pointer
301 // of the slice and at some point in the future we would
302 // overwrite it. Therefore, we must check if slice has some
303 // references in slices and if so then
304 //
305 // - we should mark those references as the original (since the data
306 // pointer should be the same)
307 //
308 // - we should make sure that the data pointer of slice
309 // does not get freed.
310 //
311 if (slice.info.state == Slice<F>::Ready) {
312 WITH_OCD WITH_RANK
313 << "__gc__:" << "checking for data recycled dependencies\n";
314 auto recycled
315 = Slice<F>::hasRecycledReferencingToIt(slices, slice.info);
316 if (recycled.size()) {
317 Slice<F>* newReady = recycled[0];
318 WITH_OCD WITH_RANK
319 << "__gc__:" << "swaping recycled "
320 << pretty_print(newReady->info)
321 << " and "
322 << pretty_print(slice.info)
323 << "\n";
324 newReady->markReady();
325 assert(newReady->data == slice.data);
326 freeSlicePointer = false;
327
328 for (size_t i = 1; i < recycled.size(); i++) {
329 auto newRecyled = recycled[i];
330 newRecyled->info.recycling = newReady->info.type;
331 WITH_OCD WITH_RANK
332 << "__gc__:" << "updating recycled "
333 << pretty_print(newRecyled->info)
334 << "\n";
335 }
336
337 }
338 }
339
340#if defined(HAVE_CUDA)
341 // In cuda, SelfSufficient slices have an ad-hoc pointer
342 // since it is a pointer on the device and has to be
343 // brought back to the free pointer bucket of the SliceUnion.
344 // Therefore, only in the Recycled case it should not be
345 // put back the pointer.
346 if (slice.info.state == Slice<F>::Recycled) {
347 freeSlicePointer = false;
348 }
349#else
350 // if the slice is self sufficient, do not dare touching the
351 // pointer, since it is a pointer to our sources in our rank.
352 if ( slice.info.state == Slice<F>::SelfSufficient
353 || slice.info.state == Slice<F>::Recycled
354 ) {
355 freeSlicePointer = false;
356 }
357#endif
358
359 // make sure we get its data pointer to be used later
360 // only for non-recycled, since it can be that we need
361 // for next iteration the data of the slice that the recycled points
362 // to
363 if (freeSlicePointer) {
364 freePointers.insert(slice.data);
365 WITH_RANK << "~~~:cl(" << name << ")"
366 << " added to freePointer "
367 << pretty_print(freePointers)
368 << "\n";
369 } else {
370 WITH_OCD WITH_RANK << "__gc__:not touching the free Pointer\n";
371 }
372
373 // at this point, let us blank the slice
374 WITH_RANK << "~~~:cl(" << name << ")"
375 << " freeing up slice "
376 // TODO: make this possible because of Templates
377 // TODO: there is a deduction error here
378 // << " info " << slice.info
379 << "\n";
380 slice.free();
381 } // we did not find the slice
382
383 }

411 LOG(0,"Atrip") << "INIT SliceUnion: " << name << "\n";

413 for (auto& ptr: sliceBuffers) {
414#if defined(HAVE_CUDA)
415 const CUresult error =

428#endif
429 }
430
431 slices
432 = std::vector<Slice<F>>(2 * sliceTypes.size(), { sources[0].size() });
433 // TODO: think exactly ^------------------- about this number
434
435 // initialize the freePointers with the pointers to the buffers
436 std::transform(sliceBuffers.begin(), sliceBuffers.end(),
437 std::inserter(freePointers, freePointers.begin()),
438 [](DataPtr<F> ptr) { return ptr; });
439
440
441
442 LOG(1,"Atrip") << "#slices " << slices.size() << "\n";
443 WITH_RANK << "#slices[0] " << slices[0].size << "\n";
444 LOG(1,"Atrip") << "#sources " << sources.size() << "\n";
445 WITH_RANK << "#sources[0] " << sources[0].size() << "\n";
446 WITH_RANK << "#freePointers " << freePointers.size() << "\n";
447 LOG(1,"Atrip") << "#sliceBuffers " << sliceBuffers.size() << "\n";
448 LOG(1,"Atrip") << "GB*" << np << " "
449 << double(sources.size() + sliceBuffers.size())
450 * sources[0].size()
451 * 8 * np
452 / 1073741824.0
453 << "\n";
454 } // constructor ends

544 DataPtr<F> unwrapSlice(typename Slice<F>::Type type, ABCTuple const& abc) {
545 WITH_CRAZY_DEBUG
546 WITH_RANK << "__unwrap__:slice " << type << " w n "
547 << name
548 << " abc" << pretty_print(abc)
549 << "\n";
550 auto& slice = Slice<F>::findByTypeAbc(slices, type, abc);
551 //WITH_RANK << "__unwrap__:info " << slice.info << "\n";
552 switch (slice.info.state) {
553 case Slice<F>::Dispatched:
554 WITH_RANK << "__unwrap__:Fetch: " << &slice
555 << " info " << pretty_print(slice.info)
556 << "\n";
557 slice.unwrapAndMarkReady();
558 return slice.data;
559 break;
560 case Slice<F>::SelfSufficient:
561 WITH_RANK << "__unwrap__:SelfSufficient: " << &slice
562 << " info " << pretty_print(slice.info)
563 << "\n";
564 return slice.data;
565 break;
566 case Slice<F>::Ready:
567 WITH_RANK << "__unwrap__:READY: UNWRAPPED ALREADY" << &slice
568 << " info " << pretty_print(slice.info)
569 << "\n";
570 return slice.data;
571 break;
572 case Slice<F>::Recycled:
573 WITH_RANK << "__unwrap__:RECYCLED " << &slice
574 << " info " << pretty_print(slice.info)
575 << "\n";
576 return unwrapSlice(slice.info.recycling, abc);
577 break;
578 case Slice<F>::Fetch:
579 case Slice<F>::Acceptor:
580 throw std::domain_error("Can't unwrap an acceptor or fetch slice!");
581 break;
582 default:
583 throw std::domain_error("Unknown error unwrapping slice!");
584 }
585 return slice.data;
586 }
