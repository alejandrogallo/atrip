55std::vector<std::string> getNodeNames(MPI_Comm comm){
56 int rank, np;
57 MPI_Comm_rank(comm, &rank);
58 MPI_Comm_size(comm, &np);
59
60 std::vector<std::string> nodeList(np);
61 char nodeName[MPI_MAX_PROCESSOR_NAME];
62 char *nodeNames = (char*)malloc(np * MPI_MAX_PROCESSOR_NAME);
63 std::vector<int> nameLengths(np)
64 , off(np)
65 ;
66 int nameLength;
67 MPI_Get_processor_name(nodeName, &nameLength);
68 MPI_Allgather(&nameLength,
69 1,
70 MPI_INT,
71 nameLengths.data(),
72 1,
73 MPI_INT,
74 comm);
75 for (int i(1); i < np; i++)
76 off[i] = off[i-1] + nameLengths[i-1];
77 MPI_Allgatherv(nodeName,
78 nameLengths[rank],
79 MPI_BYTE,
80 nodeNames,
81 nameLengths.data(),
82 off.data(),
83 MPI_BYTE,
84 comm);
85 for (int i(0); i < np; i++) {
86 std::string const s(&nodeNames[off[i]], nameLengths[i]);
87 nodeList[i] = s;
88 }
89 std::free(nodeNames);
90 return nodeList;
91}

103template <typename A>
104A unique(A const &xs) {
105 auto result = xs;
106 std::sort(std::begin(result), std::end(result));
107 auto const& last = std::unique(std::begin(result), std::end(result));
108 result.erase(last, std::end(result));
109 return result;
110}

112std::vector<RankInfo>
113getNodeInfos(std::vector<string> const& nodeNames) {
114 std::vector<RankInfo> result;
115 auto const uniqueNames = unique(nodeNames);
116 auto const index = [&uniqueNames](std::string const& s) {
117 auto const& it = std::find(uniqueNames.begin(), uniqueNames.end(), s);
118 return std::distance(uniqueNames.begin(), it);
119 };
120 std::vector<size_t> localRanks(uniqueNames.size(), 0);
121 size_t globalRank = 0;
122 for (auto const& name: nodeNames) {
123 const size_t nodeId = index(name);
124 result.push_back({name,
125 nodeId,
126 globalRank++,
127 localRanks[nodeId]++,
128 (size_t)
129 std::count(nodeNames.begin(),
130 nodeNames.end(),
131 name)
132 });
133 }
134 return result;
135}

157// [[file:~/cuda/atrip/atrip.org::*Naive%20list][Naive list:1]]
158ABCTuples getTuplesList(size_t Nv, size_t rank, size_t np) {
159
160 const size_t
161 // total number of tuples for the problem
162 n = Nv * (Nv + 1) * (Nv + 2) / 6 - Nv
163
164 // all ranks should have the same number of tuples_per_rank
165 , tuples_per_rank = n / np + size_t(n % np != 0)
166
167 // start index for the global tuples list
168 , start = tuples_per_rank * rank
169
170 // end index for the global tuples list
171 , end = tuples_per_rank * (rank + 1)
172 ;
173
174 LOG(1,"Atrip") << "tuples_per_rank = " << tuples_per_rank << "\n";
175 WITH_RANK << "start, end = " << start << ", " << end << "\n";
176 ABCTuples result(tuples_per_rank, FAKE_TUPLE);
177
178 for (size_t a(0), r(0), g(0); a < Nv; a++)
179 for (size_t b(a); b < Nv; b++)
180 for (size_t c(b); c < Nv; c++){
181 if ( a == b && b == c ) continue;
182 if ( start <= g && g < end) result[r++] = {a, b, c};
183 g++;
184 }
185
186 return result;
187
188}

231// tuple are located
232std::vector<size_t> getTupleNodes(ABCTuple const& t, size_t nNodes) {
233 std::vector<size_t>
234 nTuple = { isOnNode(t[0], nNodes)
235 , isOnNode(t[1], nNodes)
236 , isOnNode(t[2], nNodes)

247// [[file:~/cuda/atrip/atrip.org::*Distribution][Distribution:1]]
248ABCTuples specialDistribution(Info const& info, ABCTuples const& allTuples) {
249
250 ABCTuples nodeTuples;
251 size_t const nNodes(info.nNodes);
252
253 std::vector<ABCTuples>
254 container1d(nNodes)
255 , container2d(nNodes * nNodes)
256 , container3d(nNodes * nNodes * nNodes)
257 ;
258
259 WITH_DBG if (info.nodeId == 0)
260 std::cout << "\tGoing through all "
261 << allTuples.size()
262 << " tuples in "
263 << nNodes
264 << " nodes\n";
265
266 // build container-n-d's
267 for (auto const& t: allTuples) {
268 // one which node(s) are the tuple elements located...
269 // put them into the right container
270 auto const _nodes = getTupleNodes(t, nNodes);
271
272 switch (_nodes.size()) {
273 case 1:
274 container1d[_nodes[0]].push_back(t);
275 break;
276 case 2:
277 container2d[ _nodes[0]
278 + _nodes[1] * nNodes
279 ].push_back(t);
280 break;
281 case 3:
282 container3d[ _nodes[0]
283 + _nodes[1] * nNodes
284 + _nodes[2] * nNodes * nNodes
285 ].push_back(t);
286 break;
287 }
288
289 }
290
291 WITH_DBG if (info.nodeId == 0)
292 std::cout << "\tBuilding 1-d containers\n";
293 // DISTRIBUTE 1-d containers
294 // every tuple which is only located at one node belongs to this node
295 {
296 auto const& _tuples = container1d[info.nodeId];
297 nodeTuples.resize(_tuples.size(), INVALID_TUPLE);
298 std::copy(_tuples.begin(), _tuples.end(), nodeTuples.begin());
299 }
300
301 WITH_DBG if (info.nodeId == 0)
302 std::cout << "\tBuilding 2-d containers\n";
303 // DISTRIBUTE 2-d containers
304 //the tuples which are located at two nodes are half/half given to these nodes
305 for (size_t yx = 0; yx < container2d.size(); yx++) {
306
307 auto const& _tuples = container2d[yx];
308 const
309 size_t idx = yx % nNodes
310 // remeber: yx = idy * nNodes + idx
311 , idy = yx / nNodes
312 , n_half = _tuples.size() / 2
313 , size = nodeTuples.size()
314 ;
315
316 size_t nbeg, nend;
317 if (info.nodeId == idx) {
318 nbeg = 0 * n_half;
319 nend = n_half;
320 } else if (info.nodeId == idy) {
321 nbeg = 1 * n_half;
322 nend = _tuples.size();
323 } else {
324 // either idx or idy is my node
325 continue;
326 }
327
328 size_t const nextra = nend - nbeg;
329 nodeTuples.resize(size + nextra, INVALID_TUPLE);
330 std::copy(_tuples.begin() + nbeg,
331 _tuples.begin() + nend,
332 nodeTuples.begin() + size);
333
334 }
335
336 WITH_DBG if (info.nodeId == 0)
337 std::cout << "\tBuilding 3-d containers\n";
338 // DISTRIBUTE 3-d containers
339 for (size_t zyx = 0; zyx < container3d.size(); zyx++) {
340 auto const& _tuples = container3d[zyx];
341
342 const
343 size_t idx = zyx % nNodes
344 , idy = (zyx / nNodes) % nNodes
345 // remember: zyx = idx + idy * nNodes + idz * nNodes^2
346 , idz = zyx / nNodes / nNodes
347 , n_third = _tuples.size() / 3
348 , size = nodeTuples.size()
349 ;
350
351 size_t nbeg, nend;
352 if (info.nodeId == idx) {
353 nbeg = 0 * n_third;
354 nend = 1 * n_third;
355 } else if (info.nodeId == idy) {
356 nbeg = 1 * n_third;
357 nend = 2 * n_third;
358 } else if (info.nodeId == idz) {
359 nbeg = 2 * n_third;
360 nend = _tuples.size();
361 } else {
362 // either idx or idy or idz is my node
363 continue;
364 }
365
366 size_t const nextra = nend - nbeg;
367 nodeTuples.resize(size + nextra, INVALID_TUPLE);
368 std::copy(_tuples.begin() + nbeg,
369 _tuples.begin() + nend,
370 nodeTuples.begin() + size);
371
372 }
373
374
375 WITH_DBG if (info.nodeId == 0) std::cout << "\tswapping tuples...\n";
376 /*
377 * sort part of group-and-sort algorithm
378 * every tuple on a given node is sorted in a way that
379 * the 'home elements' are the fastest index.
380 * 1:yyy 2:yyn(x) 3:yny(x) 4:ynn(x) 5:nyy 6:nyn(x) 7:nny 8:nnn
381 */
382 for (auto &nt: nodeTuples){
383 if ( isOnNode(nt[0], nNodes) == info.nodeId ){ // 1234
384 if ( isOnNode(nt[2], nNodes) != info.nodeId ){ // 24
385 size_t const x(nt[0]);
386 nt[0] = nt[2]; // switch first and last
387 nt[2] = x;
388 }
389 else if ( isOnNode(nt[1], nNodes) != info.nodeId){ // 3
390 size_t const x(nt[0]);
391 nt[0] = nt[1]; // switch first two
392 nt[1] = x;
393 }
394 } else {
395 if ( isOnNode(nt[1], nNodes) == info.nodeId // 56
396 && isOnNode(nt[2], nNodes) != info.nodeId
397 ) { // 6
398 size_t const x(nt[1]);
399 nt[1] = nt[2]; // switch last two
400 nt[2] = x;
401 }
402 }
403 }
404
405 WITH_DBG if (info.nodeId == 0) std::cout << "\tsorting list of tuples...\n";
406 //now we sort the list of tuples
407 std::sort(nodeTuples.begin(), nodeTuples.end());
408
409 WITH_DBG if (info.nodeId == 0) std::cout << "\trestoring tuples...\n";
410 // we bring the tuples abc back in the order a<b<c
411 for (auto &t: nodeTuples) std::sort(t.begin(), t.end());
412
413#if ATRIP_DEBUG > 1
414 WITH_DBG if (info.nodeId == 0)
415 std::cout << "checking for validity of " << nodeTuples.size() << std::endl;
416 const bool anyInvalid
417 = std::any_of(nodeTuples.begin(),
418 nodeTuples.end(),
419 [](ABCTuple const& t) { return t == INVALID_TUPLE; });
420 if (anyInvalid) throw "Some tuple is invalid in group-and-sort algorithm";
421#endif
422
423 WITH_DBG if (info.nodeId == 0) std::cout << "\treturning tuples...\n";
424 return nodeTuples;
425
426}

429// [[file:~/cuda/atrip/atrip.org::*Main][Main:1]]
430std::vector<ABCTuple> main(MPI_Comm universe, size_t Nv) {
431
432 int rank, np;
433 MPI_Comm_rank(universe, &rank);
434 MPI_Comm_size(universe, &np);
435
436 std::vector<ABCTuple> result;
437
438 auto const nodeNames(getNodeNames(universe));
439 size_t const nNodes = unique(nodeNames).size();
440 auto const nodeInfos = getNodeInfos(nodeNames);
441
442 // We want to construct a communicator which only contains of one
443 // element per node
444 bool const computeDistribution
445 = nodeInfos[rank].localRank == 0;
446
447 std::vector<ABCTuple>
448 nodeTuples
449 = computeDistribution
450 ? specialDistribution(Info{nNodes, nodeInfos[rank].nodeId},
451 getAllTuplesList(Nv))
452 : std::vector<ABCTuple>()
453 ;
454
455 LOG(1,"Atrip") << "got nodeTuples\n";
456
457 // now we have to send the data from **one** rank on each node
458 // to all others ranks of this node
459 const
460 int color = nodeInfos[rank].nodeId
461 , key = nodeInfos[rank].localRank
462 ;
463
464
465 MPI_Comm INTRA_COMM;
466 MPI_Comm_split(universe, color, key, &INTRA_COMM);
467// Main:1 ends here
468
469// [[file:~/cuda/atrip/atrip.org::*Main][Main:2]]
470size_t const
471 tuplesPerRankLocal
472 = nodeTuples.size() / nodeInfos[rank].ranksPerNode
473 + size_t(nodeTuples.size() % nodeInfos[rank].ranksPerNode != 0)
474 ;
475
476size_t tuplesPerRankGlobal;
477
478MPI_Reduce(&tuplesPerRankLocal,
479 &tuplesPerRankGlobal,
480 1,
481 MPI_UINT64_T,
482 MPI_MAX,
483 0,
484 universe);
485
486MPI_Bcast(&tuplesPerRankGlobal,
487 1,
488 MPI_UINT64_T,
489 0,
490 universe);
491
492LOG(1,"Atrip") << "Tuples per rank: " << tuplesPerRankGlobal << "\n";
493LOG(1,"Atrip") << "ranks per node " << nodeInfos[rank].ranksPerNode << "\n";
494LOG(1,"Atrip") << "#nodes " << nNodes << "\n";
495// Main:2 ends here
496
497// [[file:~/cuda/atrip/atrip.org::*Main][Main:3]]
498size_t const totalTuples
499 = tuplesPerRankGlobal * nodeInfos[rank].ranksPerNode;
500
501if (computeDistribution) {
502 // pad with FAKE_TUPLEs
503 nodeTuples.insert(nodeTuples.end(),
504 totalTuples - nodeTuples.size(),
505 FAKE_TUPLE);
506}
507// Main:3 ends here
508
509// [[file:~/cuda/atrip/atrip.org::*Main][Main:4]]
510{
511 // construct mpi type for abctuple
512 MPI_Datatype MPI_ABCTUPLE;
513 MPI_Type_vector(nodeTuples[0].size(), 1, 1, MPI_UINT64_T, &MPI_ABCTUPLE);
514 MPI_Type_commit(&MPI_ABCTUPLE);
515
516 LOG(1,"Atrip") << "scattering tuples \n";
517
518 result.resize(tuplesPerRankGlobal);
519 MPI_Scatter(nodeTuples.data(),
520 tuplesPerRankGlobal,
521 MPI_ABCTUPLE,
522 result.data(),
523 tuplesPerRankGlobal,
524 MPI_ABCTUPLE,
525 0,
526 INTRA_COMM);
527
528 MPI_Type_free(&MPI_ABCTUPLE);
529
530}
531// Main:4 ends here
532
533// [[file:~/cuda/atrip/atrip.org::*Main][Main:5]]
534return result;
535
536}
