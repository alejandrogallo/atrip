41 
42  const size_t
43  // total number of tuples for the problem
44  n = Nv * (Nv + 1) * (Nv + 2) / 6 - Nv
45 
46  // all ranks should have the same number of tuples_per_rank
47  , tuples_per_rank = n / np + size_t(n % np != 0)
48 
49  // start index for the global tuples list

53  , end = tuples_per_rank * (rank + 1)

56  ABCTuples result(tuples_per_rank, FAKE_TUPLE);

60  for (size_t c(b); c < Nv; c++, g++){
61  if ( a == b && b == c ) continue;
62  if ( g > start && g <= end) result[r++] = {a, b, c};
