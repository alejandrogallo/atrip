104 A unique(A const &xs) {
105  auto result = xs;
106  std::sort(std::begin(result), std::end(result));
107  auto const& last = std::unique(std::begin(result), std::end(result));
108  result.erase(last, std::end(result));
109  return result;
110 }

112 std::vector<RankInfo>
113 getNodeInfos(std::vector<string> const& nodeNames) {
114  std::vector<RankInfo> result;
115  auto const uniqueNames = unique(nodeNames);
116  auto const index = [&uniqueNames](std::string const& s) {
117  auto const& it = std::find(uniqueNames.begin(), uniqueNames.end(), s);
118  return std::distance(uniqueNames.begin(), it);
119  };
120  std::vector<size_t> localRanks(uniqueNames.size(), 0);
121  size_t globalRank = 0;
122  for (auto const& name: nodeNames) {
123  const size_t nodeId = index(name);
124  result.push_back({name,
125  nodeId,
126  globalRank++,
127  localRanks[nodeId]++,
128  std::count(nodeNames.begin(),
129  nodeNames.end(),
130  name)
131  });
132  }
133  return result;
134 }

157 ABCTuples getTuplesList(size_t Nv, size_t rank, size_t np) {
158 
159  const size_t
160  // total number of tuples for the problem
161  n = Nv * (Nv + 1) * (Nv + 2) / 6 - Nv
162 
163  // all ranks should have the same number of tuples_per_rank
164  , tuples_per_rank = n / np + size_t(n % np != 0)
165 
166  // start index for the global tuples list
167  , start = tuples_per_rank * rank
168 
169  // end index for the global tuples list
170  , end = tuples_per_rank * (rank + 1)
171  ;
172 
173  LOG(1,"Atrip") << "tuples_per_rank = " << tuples_per_rank << "\n";
174  WITH_RANK << "start, end = " << start << ", " << end << "\n";
175  ABCTuples result(tuples_per_rank, FAKE_TUPLE);
176 
177  for (size_t a(0), r(0), g(0); a < Nv; a++)
178  for (size_t b(a); b < Nv; b++)
179  for (size_t c(b); c < Nv; c++){
180  if ( a == b && b == c ) continue;
181  if ( start <= g && g < end) result[r++] = {a, b, c};
182  g++;
183  }
184 
185  return result;
186 
187 }

247 ABCTuples specialDistribution(Info const& info, ABCTuples const& allTuples) {
248 
249  ABCTuples nodeTuples;
250  size_t const nNodes(info.nNodes);
251 
252  std::vector<ABCTuples>
253  container1d(nNodes)
254  , container2d(nNodes * nNodes)
255  , container3d(nNodes * nNodes * nNodes)
256  ;
257 
258  WITH_DBG if (info.nodeId == 0)
259  std::cout << "\tGoing through all "
260  << allTuples.size()
261  << " tuples in "
262  << nNodes
263  << " nodes\n";
264 
265  // build container-n-d's
266  for (auto const& t: allTuples) {
267  // one which node(s) are the tuple elements located...
268  // put them into the right container
269  auto const _nodes = getTupleNodes(t, nNodes);
270 
271  switch (_nodes.size()) {
272  case 1:
273  container1d[_nodes[0]].push_back(t);
274  break;
275  case 2:
276  container2d[ _nodes[0]
277  + _nodes[1] * nNodes
278  ].push_back(t);
279  break;
280  case 3:
281  container3d[ _nodes[0]
282  + _nodes[1] * nNodes
283  + _nodes[2] * nNodes * nNodes
284  ].push_back(t);
285  break;
286  }
287 
288  }
289 
290  WITH_DBG if (info.nodeId == 0)
291  std::cout << "\tBuilding 1-d containers\n";
292  // DISTRIBUTE 1-d containers
293  // every tuple which is only located at one node belongs to this node
294  {
295  auto const& _tuples = container1d[info.nodeId];
296  nodeTuples.resize(_tuples.size(), INVALID_TUPLE);
297  std::copy(_tuples.begin(), _tuples.end(), nodeTuples.begin());
298  }
299 
300  WITH_DBG if (info.nodeId == 0)
301  std::cout << "\tBuilding 2-d containers\n";
302  // DISTRIBUTE 2-d containers
303  //the tuples which are located at two nodes are half/half given to these nodes
304  for (size_t yx = 0; yx < container2d.size(); yx++) {
305 
306  auto const& _tuples = container2d[yx];
307  const
308  size_t idx = yx % nNodes
309  // remeber: yx = idy * nNodes + idx
310  , idy = yx / nNodes
311  , n_half = _tuples.size() / 2
312  , size = nodeTuples.size()
313  ;
314 
315  size_t nbeg, nend;
316  if (info.nodeId == idx) {
317  nbeg = 0 * n_half;
318  nend = n_half;
319  } else if (info.nodeId == idy) {
320  nbeg = 1 * n_half;
321  nend = _tuples.size();
322  } else {
323  // either idx or idy is my node
324  continue;
325  }
326 
327  size_t const nextra = nend - nbeg;
328  nodeTuples.resize(size + nextra, INVALID_TUPLE);
329  std::copy(_tuples.begin() + nbeg,
330  _tuples.begin() + nend,
331  nodeTuples.begin() + size);
332 
333  }
334 
335  WITH_DBG if (info.nodeId == 0)
336  std::cout << "\tBuilding 3-d containers\n";
337  // DISTRIBUTE 3-d containers
338  for (size_t zyx = 0; zyx < container3d.size(); zyx++) {
339  auto const& _tuples = container3d[zyx];
340 
341  const
342  size_t idx = zyx % nNodes
343  , idy = (zyx / nNodes) % nNodes
344  // remember: zyx = idx + idy * nNodes + idz * nNodes^2
345  , idz = zyx / nNodes / nNodes
346  , n_third = _tuples.size() / 3
347  , size = nodeTuples.size()
348  ;
349 
350  size_t nbeg, nend;
351  if (info.nodeId == idx) {
352  nbeg = 0 * n_third;
353  nend = 1 * n_third;
354  } else if (info.nodeId == idy) {
355  nbeg = 1 * n_third;
356  nend = 2 * n_third;
357  } else if (info.nodeId == idz) {
358  nbeg = 2 * n_third;
359  nend = _tuples.size();
360  } else {
361  // either idx or idy or idz is my node
362  continue;
363  }
364 
365  size_t const nextra = nend - nbeg;
366  nodeTuples.resize(size + nextra, INVALID_TUPLE);
367  std::copy(_tuples.begin() + nbeg,
368  _tuples.begin() + nend,
369  nodeTuples.begin() + size);
370 
371  }
372 
373 
374  WITH_DBG if (info.nodeId == 0) std::cout << "\tswapping tuples...\n";
375  /*
376  * sort part of group-and-sort algorithm
377  * every tuple on a given node is sorted in a way that
378  * the 'home elements' are the fastest index.
379  * 1:yyy 2:yyn(x) 3:yny(x) 4:ynn(x) 5:nyy 6:nyn(x) 7:nny 8:nnn
380  */
381  for (auto &nt: nodeTuples){
382  if ( isOnNode(nt[0], nNodes) == info.nodeId ){ // 1234
383  if ( isOnNode(nt[2], nNodes) != info.nodeId ){ // 24
384  size_t const x(nt[0]);
385  nt[0] = nt[2]; // switch first and last
386  nt[2] = x;
387  }
388  else if ( isOnNode(nt[1], nNodes) != info.nodeId){ // 3
389  size_t const x(nt[0]);
390  nt[0] = nt[1]; // switch first two
391  nt[1] = x;
392  }
393  } else {
394  if ( isOnNode(nt[1], nNodes) == info.nodeId // 56
395  && isOnNode(nt[2], nNodes) != info.nodeId
396  ) { // 6
397  size_t const x(nt[1]);
398  nt[1] = nt[2]; // switch last two
399  nt[2] = x;
400  }
401  }
402  }
403 
404  WITH_DBG if (info.nodeId == 0) std::cout << "\tsorting list of tuples...\n";
405  //now we sort the list of tuples
406  std::sort(nodeTuples.begin(), nodeTuples.end());
407 
408  WITH_DBG if (info.nodeId == 0) std::cout << "\trestoring tuples...\n";
409  // we bring the tuples abc back in the order a<b<c
410  for (auto &t: nodeTuples) std::sort(t.begin(), t.end());
411 
412 #if ATRIP_DEBUG > 1
413  WITH_DBG if (info.nodeId == 0)
414  std::cout << "checking for validity of " << nodeTuples.size() << std::endl;
415  const bool anyInvalid
416  = std::any_of(nodeTuples.begin(),
417  nodeTuples.end(),
418  [](ABCTuple const& t) { return t == INVALID_TUPLE; });
419  if (anyInvalid) throw "Some tuple is invalid in group-and-sort algorithm";
420 #endif
421 
422  WITH_DBG if (info.nodeId == 0) std::cout << "\treturning tuples...\n";
423  return nodeTuples;
424 
425 }

429 std::vector<ABCTuple> main(MPI_Comm universe, size_t Nv) {
430

434

436

438  size_t const nNodes = unique(nodeNames).size();
439  auto const nodeInfos = getNodeInfos(nodeNames);
440 
441  // We want to construct a communicator which only contains of one
442  // element per node
443  bool const computeDistribution
444  = nodeInfos[rank].localRank == 0;
445 
446  std::vector<ABCTuple>
447  nodeTuples
448  = computeDistribution

459  int color = nodeInfos[rank].nodeId
460  , key = nodeInfos[rank].localRank
461  ;
462 
463
