116 static MPI_Datatype vector(size_t n, MPI_Datatype const& DT) {

118 MPI_Type_vector(n, 1, 1, DT, &dt);

120 return dt;

123 static MPI_Datatype sliceLocation () {
124 constexpr int n = 2;
125 // create a sliceLocation to measure in the current architecture
126 // the packing of the struct
127 Slice<F>::Location measure;
128 MPI_Datatype dt;
129 const std::vector<int> lengths(n, 1);
130 const MPI_Datatype types[n] = {usizeDt(), usizeDt()};
131
132 static_assert(sizeof(Slice<F>::Location) == 2 * sizeof(size_t),
133 "The Location packing is wrong in your compiler");
134
135 // measure the displacements in the struct
136 size_t j = 0;
137 MPI_Aint base_address, displacements[n];
138 MPI_Get_address(&measure, &base_address);
139 MPI_Get_address(&measure.rank, &displacements[j++]);
140 MPI_Get_address(&measure.source, &displacements[j++]);
141 for (size_t i = 0; i < n; i++)
142 displacements[i] = MPI_Aint_diff(displacements[i], base_address);
143
144 MPI_Type_create_struct(n, lengths.data(), displacements, types, &dt);
145 MPI_Type_commit(&dt);
146 return dt;
147 }
148
149 static MPI_Datatype usizeDt() { return MPI_UINT64_T; }
150
151 static MPI_Datatype sliceInfo () {
152 constexpr int n = 5;
153 MPI_Datatype dt;
154 Slice<F>::Info measure;
155 const std::vector<int> lengths(n, 1);
156 const MPI_Datatype types[n]
157 = { vector(2, usizeDt())
158 , vector(sizeof(enum Type), MPI_CHAR)
159 , vector(sizeof(enum State), MPI_CHAR)
160 , sliceLocation()
161 , vector(sizeof(enum Type), MPI_CHAR)
162 // TODO: Why this does not work on intel mpi?
163 /*, MPI_UINT64_T*/
164 };
165
166 static_assert(sizeof(enum Type) == 4, "Enum type not 4 bytes long");
167 static_assert(sizeof(enum State) == 4, "Enum State not 4 bytes long");
168 static_assert(sizeof(enum Name) == 4, "Enum Name not 4 bytes long");
169
170 // create the displacements from the info measurement struct
171 size_t j = 0;
172 MPI_Aint base_address, displacements[n];
173 MPI_Get_address(&measure, &base_address);
174 MPI_Get_address(&measure.tuple[0], &displacements[j++]);
175 MPI_Get_address(&measure.type, &displacements[j++]);
176 MPI_Get_address(&measure.state, &displacements[j++]);
177 MPI_Get_address(&measure.from, &displacements[j++]);
178 MPI_Get_address(&measure.recycling, &displacements[j++]);
179 for (size_t i = 0; i < n; i++)
180 displacements[i] = MPI_Aint_diff(displacements[i], base_address);
181
182 MPI_Type_create_struct(n, lengths.data(), displacements, types, &dt);
183 MPI_Type_commit(&dt);
184 return dt;
185 }

220static
221PartialTuple subtupleBySlice(ABCTuple abc, Type sliceType) {
222 switch (sliceType) {
223 case AB: return {abc[0], abc[1]};
224 case BC: return {abc[1], abc[2]};
225 case AC: return {abc[0], abc[2]};
226 case CB: return {abc[2], abc[1]};
227 case BA: return {abc[1], abc[0]};
228 case CA: return {abc[2], abc[0]};
229 case A: return {abc[0], 0};
230 case B: return {abc[1], 0};
231 case C: return {abc[2], 0};
232 default: throw "Switch statement not exhaustive!";
233 }
234}

437void unwrapAndMarkReady() {
438 if (info.state == Ready) return;
439 if (info.state != Dispatched)

442 markReady();
443 MPI_Status status;
444#ifdef HAVE_OCD
445 WITH_RANK << "__slice__:mpi: waiting " << "\n";
446#endif
447 const int errorCode = MPI_Wait(&request, &status);
448
449 // FIXME: it appears not to work to free
450 // this request, investigate if this is necessary or not
451 //const auto _mpi_request_free = MPI_Request_free(&request);
452
453 //if (MPI_SUCCESS != _mpi_request_free)
454 //throw "Atrip: Error freeing MPI request";
455
456 if (errorCode != MPI_SUCCESS)
457 throw "Atrip: Unexpected error MPI ERROR";
458
459#if defined(HAVE_CUDA)
460 // copy the retrieved mpi data to the device
461 WITH_CHRONO("cuda:memcpy",
462 _CHECK_CUDA_SUCCESS("copying mpi data to device",
463 cuMemcpyHtoD(data,

465 sizeof(F) * size));
466 )
467 std::free(mpi_data);
468#endif

470#ifdef HAVE_OCD

473 MPI_Error_string(errorCode, errorString, &errorSize);
