135 static MPI_Datatype sliceLocation () {
136 constexpr int n = 2;
137 // create a sliceLocation to measure in the current architecture
138 // the packing of the struct
139 Slice<F>::Location measure;
140 MPI_Datatype dt;
141 const std::vector<int> lengths(n, 1);
142 const MPI_Datatype types[n] = {usizeDt(), usizeDt()};
143
144 static_assert(sizeof(Slice<F>::Location) == 2 * sizeof(size_t),
145 "The Location packing is wrong in your compiler");
146
147 // measure the displacements in the struct
148 size_t j = 0;
149 MPI_Aint base_address, displacements[n];
150 MPI_Get_address(&measure, &base_address);
151 MPI_Get_address(&measure.rank, &displacements[j++]);
152 MPI_Get_address(&measure.source, &displacements[j++]);
153 for (size_t i = 0; i < n; i++)
154 displacements[i] = MPI_Aint_diff(displacements[i], base_address);
155
156 MPI_Type_create_struct(n, lengths.data(), displacements, types, &dt);
157 MPI_Type_commit(&dt);
158 return dt;
159 }
160
161 static MPI_Datatype usizeDt() { return MPI_UINT64_T; }
162
163 static MPI_Datatype sliceInfo () {
164 constexpr int n = 5;
165 MPI_Datatype dt;
166 Slice<F>::Info measure;
167 const std::vector<int> lengths(n, 1);
168 const MPI_Datatype types[n]
169 = { vector(2, usizeDt())
170 , vector(sizeof(enum Type), MPI_CHAR)
171 , vector(sizeof(enum State), MPI_CHAR)
172 , sliceLocation()
173 , vector(sizeof(enum Type), MPI_CHAR)
174 // TODO: Why this does not work on intel mpi?
175 /*, MPI_UINT64_T*/
176 };
177
178 static_assert(sizeof(enum Type) == 4, "Enum type not 4 bytes long");
179 static_assert(sizeof(enum State) == 4, "Enum State not 4 bytes long");
180 static_assert(sizeof(enum Name) == 4, "Enum Name not 4 bytes long");
181
182 // create the displacements from the info measurement struct
183 size_t j = 0;
184 MPI_Aint base_address, displacements[n];
185 MPI_Get_address(&measure, &base_address);
186 MPI_Get_address(&measure.tuple[0], &displacements[j++]);
187 MPI_Get_address(&measure.type, &displacements[j++]);
188 MPI_Get_address(&measure.state, &displacements[j++]);
189 MPI_Get_address(&measure.from, &displacements[j++]);
190 MPI_Get_address(&measure.recycling, &displacements[j++]);
191 for (size_t i = 0; i < n; i++)
192 displacements[i] = MPI_Aint_diff(displacements[i], base_address);
193
194 MPI_Type_create_struct(n, lengths.data(), displacements, types, &dt);
195 MPI_Type_commit(&dt);
196 return dt;
197 }

232static
233PartialTuple subtupleBySlice(ABCTuple abc, Type sliceType) {
234 switch (sliceType) {
235 case AB: return {abc[0], abc[1]};
236 case BC: return {abc[1], abc[2]};
237 case AC: return {abc[0], abc[2]};
238 case CB: return {abc[2], abc[1]};
239 case BA: return {abc[1], abc[0]};
240 case CA: return {abc[2], abc[0]};
241 case A: return {abc[0], 0};
242 case B: return {abc[1], 0};
243 case C: return {abc[2], 0};
244 default: throw "Switch statement not exhaustive!";
245 }
246}
